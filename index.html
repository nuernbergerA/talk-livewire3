class EditPost extends Component
{
  public Post $post;

  protected $rules = [
    'post.title' => 'required|string|min:6',
    'post.content' => 'required|string|max:500',
  ];

  public function save()
  {
    $this->validate();
    $this->post->save();
  }
}
